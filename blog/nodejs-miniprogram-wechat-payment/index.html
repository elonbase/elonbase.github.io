# 前言

前段时间给朋友做了个小程序，在微信支付部分踩了许多坑，现在再来复习一遍吧！

该小程序是一个线上便利商店，实现对商品的管理，订单管理，订单支付等功能。

本篇文章将展示如何开发微信小程序支付功能，如何使用Node.js与微信支付服务进行通信，以及该过程的数据处理。

var xml2js = require('xml2js')
var request = require('request')
var crypto = require('crypto')
var queryString = require('querystring')

payOrder(req, res, next) {
  let order = req.body
  // 生成统一下单接口参数
  let unifiedorderParams = {
    appid: config.appid,            // 小程序的appid
    body: 'xx便利店-佳洁士牙膏600ml',  // 商品描述
    mch_id: config.mch_id,          // 商户ID
    nonce_str: createNonceStr(),    // 随机字符串
    notify_url: config.notify_url,  // 微信付款后的回调地址
    out_trade_no: order.order_no,   // 商户订单号
    total_fee: order.total_fee,     // 订单金额（以分为单位）
    trade_type: "JSAPI",
    openid: order.openid            // 用户标识
  };
  // 获取 sign 参数
  unifiedorderParams.sign = getSign(unifiedorderParams);
  // 将参数生成xml
  let paymentBody = buildXML(unifiedorderParams)
  // 向微信发起下单请求
  request.post({ url: config.unifiedorder_url, body: paymentBody }, function (error, response, body) {
    if (!error && response.statusCode == 200) {
      // 微信返回的数据为 xml 格式， 需要装换为 json 数据， 便于使用
      xml2js.parseStringPromise(body).then(function(result) {
        let xmlData = result.xml
        if (xmlData.return_code[0] == 'SUCCESS' && xmlData.prepay_id[0]) {
          //下单成功
          let prepay_id = xmlData.prepay_id[0]
          // 组装小程序端用于发起微信支付的参数
          let paymentParams = {
            appId: config.appid, 
            timeStamp: parseInt(new Date().getTime() / 1000).toString(), //时间戳，自1970年以来的秒数
            nonceStr: createNonceStr(),
            package: "prepay_id=" + prepay_id,  //预支付交易会话标识 
            signType: "MD5"
          }
          paymentParams.paySign = getSign(paymentParams)
          res.end(paymentParams)
        } else {
          res.end(xmlData)
        }
      }).catch(parseError => {
        res.end(parseError)
      })
    } else {
      res.end(body);
    }
  }).on('error', function(postError) {
    res.end(postError)
  })
}

function createNonceStr() {
  let str = Math.random().toString(36).substr(2, 15)
  return str
}

function sortObj(obj) {
  // 按参数名ASCII码从小到大排序
  var keys = Object.keys(obj).sort()
  var newArgs = {};
  keys.forEach(function (val, key) {
    if (obj[val]){
      newArgs[val] = obj[val];
    }
  })
  return newArgs
}

function getSign(signParams){
  let newArgs = sortObj(signParams)
  // 拼接API密钥
  var string = queryString.stringify(newArgs)+'&key='+config.appKeyScrect
  // 生成签名
  return crypto.createHash('md5').update(queryString.unescape(string), 'utf8').digest("hex").toUpperCase();
}

function buildXML(obj) {
  let newObj = sortObj(obj)
  var builder = new xml2js.Builder();
  var xml = builder.buildObject({xml:newObj})
  return xml;
}

微信提供了微信支付仿真测试系统sandbox (opens new window)，我在使用sandbox做支付测试时，一直是签名失败，没有成功过。后面直接用生产环境, 就正常了。我不知道是哪里漏掉了什么，或是仿真系统存在bug,后面没去验证。所以，这里建议，直接用生产环境做测试吧。

首先，组装unifiedorderParams对象，然后使用该对象格式化为URL键值对的格式(key1=value1&key2=value2)，在得到的字符串后面拼接商户密钥 (opens new window)（微信商户平台(pay.weixin.qq.com)-->账户设置-->API安全-->密钥设置），将最终得到字符串进行MD5加密，将加密后得到的字符串赋值给unifiedorderParams的sign属性。最后将unifiedorderParams对象格式化为XML。这时，统一下单的参数才算组装完成。可参见安全规范 (opens new window)

# 3.发起微信支付

如果一切顺利，小程序端的unifiedOrder将收到Node接口(https://younodeservice.com/order/payment')返回的参数，其中包含用于发起微信支付的prepay_id。

接下来，由小程序端发起微信支付 (opens new window)

// 示例代码 - 小程序端
const requestWXPayment = params => {
  wx.requestPayment({
    ...params,
    success: (res) => {
      // 支付成功处理逻辑
      wx.showToast({
        title: '订单支付成功',
      })
    },
    fail (res) {
      wx.showToast({
        title: res.errMsg
      })
    }
  })
}

# 4.支付结果通知

支付完成后，微信会把相关支付结果及用户信息通过数据流的形式发送给商户，商户需要接收处理，并按文档规范返回应答。

支付成功后，微信会调用在统一下单参数中设置的(notify_url)结果通知地址，在该接口方法中，你需要处理微信支付成功后给你返回的数据，然后并以规定的格式返回数据给微信，告诉它，你已经知道支付成功了。

// 示例代码 - Node.js
paymentSuccessCallback(req, res, next) {
    let wxPayBack = req.body.xml
    //签名验证,并校验返回的订单金额是否与商户侧的订单金额一致
    //将对微信回调参数参数重新签名并于参数中的签名对比，如果一致，修改订单参数，并返回相应参数回微信
    //否则，不对订单状态进行处理，并返回失败状态给微信
    if (wxPayBack && wxPayBack.result_code == 'SUCCESS') {
      let wxSign = wxPayBack.sign
      //使用微信返回的支付结果（除sign字段）再次签名，
      //将等到的签名与微信返回结果中的签名对比，若相等，则说明数据未被篡改
      delete wxPayBack.sign
      let theSign = getSign(wxPayBack)
      if (theSign == wxSign) {
        var json2Xml = function (json) {
          let _xml = '';
          Object.keys(json).map((key) => {
            _xml += `<${key}>${json[key]}</${key}>`;
          })
          return `<xml>${_xml}</xml>`;
        }
        var sendData = {
          return_code: 'SUCCESS',
          return_msg: 'OK'
        }
        res.end(json2Xml(sendData))
      } else {
        res.end()
      }
    } else {
      res.end()
    }
  }

注意：由于微信调用paymentSuccessCallback传的参数是XML格式的，为了方便使用，我在Node接口路由(route.js)中使用了express-xml-bodyparser中间件处理了参数数据。以便可以使用eq.body.xml获取到参数的JSON格式数据。

# 总结

微信支付的坑点，主要是在加密签名那一部分，然后就是使用XML格式做参数的传输，平时用惯了JSON，所以在处理XML时化了一些功夫。
