# 前言

微信退款，我觉得难点是在对微信退款接口返回数据要进行解密处理的部分。 这里想吐槽的是，微信退款部分的文档写得不是很清楚，在这部分花了很多时间。

本篇文章将展示如何使用Node.js打通微信退款功能，以及如何对返回的数据进行解密。

# 微信退款

  1. 根据退款申请 (opens new window)文档，组装必要的退款参数。

  var refundBody = buildXML(signParams)

function createNonceStr() {
  let str = Math.random().toString(36).substr(2, 15)
  return str
}

function sortObj(obj) {
  // 按参数名ASCII码从小到大排序
  var keys = Object.keys(obj).sort()
  var newArgs = {};
  keys.forEach(function (val, key) {
    if (obj[val]){
      newArgs[val] = obj[val];
    }
  })
  return newArgs
}

function getSign(signParams){
  let newArgs = sortObj(signParams)
  // 拼接API密钥
  var string = queryString.stringify(newArgs)+'&key='+config.appKeyScrect
  // 生成签名
  return crypto.createHash('md5').update(queryString.unescape(string), 'utf8').digest("hex").toUpperCase();
}

function buildXML(obj) {
  let newObj = sortObj(obj)
  var builder = new xml2js.Builder();
  var xml = builder.buildObject({xml:newObj})
  return xml;
}

# 退款结果通知

当商户申请的退款有结果后（退款状态为：退款成功、退款关闭、退款异常），微信会把相关结果发送给商户，商户需要接收处理，并返回应答。

退款成功后，微信会调用之前在退款申请(wxRefund)中设置的refund_notify_url接口地址，你需要在该接口方法中，你需要处理微信退款成功后给你返回的数据，然后并以规定的格式返回数据给微信，告诉它，你已经知道退款成功了。

wxRefundSuccess(req, res, next) {
  let refundBack = req.body.xml
  if (refundBack && refundBack.return_code == 'SUCCESS') {
    // 解密退款回调数据
    let data = decodeRefundData(refundBack.req_info)
    // 将最外层的root节点名替换成xml
    let reg = new RegExp("root>", "g");
    let xmlData = data.replace(reg,'xml>');

    xml2js.parseStringPromise(xmlData).then(function(result) {
      let refund_id = result.xml.refund_id[0]
      let refund_status = result.xml.refund_status[0]
      let out_trade_no = result.xml.out_trade_no[0]
      if (refund_id && refund_status == 'SUCCESS') {
        //更新订单状态并应答微信
        var json2Xml = function (json) {
          let _xml = '';
          Object.keys(json).map((key) => {
            _xml += `<${key}>${json[key]}</${key}>`;
          })
          return `<xml>${_xml}</xml>`;
        }
        var sendData = {
          return_code: 'SUCCESS',
          return_msg: 'OK'
        }
        res.end(json2Xml(sendData))
      } else {
        res.end()
        console.log('退款异常：'+JSON.stringify(xmlData))
      }
    }).catch(error => {
      console.log(error)
    })
  } else {
    res.end()
  }
}

注意：由于微信调用wxRefundSuccess传的参数是XML格式的，为了方便使用，我在Node接口路由(route.js)中使用了express-xml-bodyparser中间件处理了参数数据。以便可以使用eq.body.xml获取到参数的JSON格式数据。

# *解密返回数据

function decodeRefundData(req_info) {
  //1.对加密串A做base64解码，得到加密串B
  let decodeInfo = new Buffer.from(req_info, 'base64')

  //2.对商户key做md5，得到32位小写key* ( key设置路径：微信商户平台(pay.weixin.qq.com)-->账户设置-->API安全-->密钥设置 )
  let md5String = crypto.createHash('md5').update(config.appKeyScrect, 'utf8').digest("hex")

  //3.用key*对加密串B做AES-256-ECB解密（PKCS7Padding）
  var clearEncoding = 'utf8';
  var cipherEncoding = 'base64';
  var cipherChunks = [];
  var iv = Buffer.alloc(0)
  var decipher = crypto.createDecipheriv('aes-256-ecb', md5String, iv);
  decipher.setAutoPadding(true);
  let a = decipher.update(decodeInfo, cipherEncoding, clearEncoding)
  let b = decipher.final(clearEncoding)
  cipherChunks.push(a);
  cipherChunks.push(b);
  return cipherChunks.join('');
}
